class Solution {
    public int chalkReplacer(int[] chalk, int k) {
        int n = chalk.length;
        long sum = 0;
        for(int chalkSum : chalk){
            sum += chalkSum;
        }
       
       int num = (int)(k%sum);
      

        for(int i=0;i<n;i++){
            if(num<chalk[i]){
                return i;
            }
         num -= chalk[i];
        }
        return -1;
    }
}
